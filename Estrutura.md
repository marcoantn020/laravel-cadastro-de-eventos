## if / if-else / if-elseif-else

    ______________if____________________
    @if(condicao)
        bloco a ser executado
    @endif
    ______________if-else____________________
    @if(condicao)
        bloco a ser executado
    @else
        bloco a ser executado
    @endif
    ______________if-elseif-else____________________
    @if(condicao)
        bloco a ser executado
    @elseif
        bloco a ser executado
    @else
        bloco a ser executado
    @endif
